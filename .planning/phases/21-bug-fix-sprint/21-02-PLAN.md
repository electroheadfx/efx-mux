  e.preventDefault(); e.stopPropagation();

// Row context-menu — invoked per file/folder via right-click.
function buildOpenInChildren(path: string): ContextMenuItem[] {
  const ed = detectedEditors.value;
  if (!ed) return [];
  const children: ContextMenuItem[] = [];
  if (ed.zed)    children.push({ label: 'Zed',                action: () => { void launchOrToast('Zed', path); } });
  if (ed.code)   children.push({ label: 'Visual Studio Code', action: () => { void launchOrToast('Visual Studio Code', path); } });
  if (ed.cursor) children.push({ label: 'Cursor',             action: () => { void launchOrToast('Cursor', path); } });
  if (ed.subl)   children.push({ label: 'Sublime Text',       action: () => { void launchOrToast('Sublime Text', path); } });
  if (ed.idea)   children.push({ label: 'IntelliJ IDEA',      action: () => { void launchOrToast('IntelliJ IDEA', path); } });
  return children;
}

async function launchOrToast(app: string, path: string): Promise<void> {

**Step 3 — Add temporal instrumentation (backend):**
Inside `src-tauri/src/file_ops.rs`, modify `launch_external_editor_impl` (line 554) to (a) log entry/exit, (b) actually CHECK the spawn child's exit status, not just spawn-and-forget. The current `.spawn().map(|_| ())` discards spawn errors AND any non-zero exit. Replace with:

```rust
pub fn launch_external_editor_impl(app: &str, path: &str) -> Result<(), String> {
    eprintln!("[FIX-05] launch_external_editor_impl: app={:?} path={:?}", app, path);
    if !is_safe_path(path) {
        eprintln!("[FIX-05] launch_external_editor_impl: is_safe_path REJECTED path={:?}", path);
        return Err("Invalid path: directory traversal not allowed".to_string());
    }
    let mut child = std::process::Command::new("open")
        .args(["-a", app, path])
        .spawn()
        .map_err(|e| { eprintln!("[FIX-05] spawn failed: {:?}", e); e.to_string() })?;
    // `open -a App path` should exit immediately (LaunchServices fork). Wait briefly for status.
    match child.wait() {
        Ok(status) if status.success() => {
            eprintln!("[FIX-05] open exited 0");
            Ok(())
        }
        Ok(status) => {
            let msg = format!("open -a {:?} exited with status {:?}", app, status.code());
            eprintln!("[FIX-05] {}", msg);
            Err(msg)
        }
        Err(e) => {
            eprintln!("[FIX-05] child.wait failed: {:?}", e);
            Err(e.to_string())
        }
    }
}
```

Note: The CHANGE in behavior (waiting on `open`) IS the suspected fix. `open -a App path` returns immediately if the app launches successfully but exits non-zero if it can't find the app or if the path doesn't exist. The prior `.spawn().map(|_| ())` swallowed both successes and failures equally — making the failure look identical to success from the frontend's point of view.

If the user has Zed installed at `/Applications/Zed.app` but `detect_editors` reports `zed: true` for a different reason (e.g., binary on PATH but no .app bundle), `open -a Zed` would fail with "Unable to find application named 'Zed'". This matches the symptom.

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Frontend → IPC → `launch_external_editor` | `app` and `path` are passed as separate argv elements (no shell interpolation). `is_safe_path` validates `path`. `app` is constrained to a small whitelist via the dropdown labels (user cannot inject arbitrary app names through the UI). |
