Add `revert_file_impl(repo_path: &str, file_path: &str) -> Result<(), GitError>`:
- Open repo with `Repository::open`, error -> `GitError::NotARepo`
- Get workdir, compute `rel_path` (same absolute/relative logic as `stage_file_impl`)
- Use system `git checkout -- {rel_path}` via `std::process::Command` (same pattern as `push_impl` -- shell out to system git because git2's `checkout_index` has edge cases with submodules and permissions that system git handles correctly)
- Run from `repo_path` as current_dir
- On non-zero exit: return `GitError::RevertError(stderr)`
- On success: return `Ok(())`
- NOTE: Only revert tracked modified files. For untracked files (`?` status), this is a no-op (git checkout does nothing for untracked files, which is the correct behavior -- untracked files should not be silently deleted).

Add the async Tauri command wrapper `revert_file`:
```rust
#[tauri::command]
pub async fn revert_file(repo_path: String, file_path: String) -> Result<(), String> {
    spawn_blocking(move || revert_file_impl(&repo_path, &file_path))
        .await
        .map_err(|e| e.to_string())?
        .map_err(|e| e.to_string())
}
```

   ```typescript
   async function handleRevertFile(file: GitFile): Promise<void> {
     const project = getActiveProject();
     if (!project) return;
     try {
       await revertFile(project.path, file.path);
       await refreshGitFiles();
     } catch (err) {
       const errMsg = err instanceof GitError ? err.details || err.code : String(err);
       addLogEntry('error', `Failed to revert ${file.name}`, errMsg);
       showToast({ type: 'error', message: `Failed to revert ${file.name}`, hint: 'See error log in GIT tab' });
     }
   }

   async function handleRevertAll(): Promise<void> {
     const project = getActiveProject();
     if (!project) return;
     isReverting.value = true;
     try {
       for (const file of changedFiles.value) {
         await revertFile(project.path, file.path);
       }
       await refreshGitFiles();
     } catch (err) {
       const errMsg = err instanceof GitError ? err.details || err.code : String(err);
       addLogEntry('error', 'Failed to revert all files', errMsg);
       showToast({ type: 'error', message: 'Failed to revert all files', hint: 'See error log in GIT tab' });
     } finally {
       isReverting.value = false;
     }
   }
   ```
