import { closeSubScope } from './sub-scope-pane';
import { gsdTab } from './unified-tab-bar';
import { getTerminalScope } from './terminal-tabs';

describe('Phase 22 gap-closure (22-10): closeSubScope', () => {
  beforeEach(() => {
    __resetActiveSubScopesForTesting();
    gsdTab.value = null;
    getTerminalScope('main-0').tabs.value = [];
    getTerminalScope('main-1').tabs.value = [];
    getTerminalScope('main-2').tabs.value = [];
    getTerminalScope('right-0').tabs.value = [];
    getTerminalScope('right-1').tabs.value = [];
  });

  it('removes the scope from activeMainSubScopes (fill-gap-at-end convention)', () => {
    activeMainSubScopes.value = ['main-0', 'main-1', 'main-2'];
    closeSubScope('main', 1);
    // Per fill-gap convention, list shrinks from end (last scope id removed; tabs from removed slot migrated to scope-0)
    expect(activeMainSubScopes.value).toEqual(['main-0', 'main-1']);
  });

  it('migrates tabs from closed scope to scope-0', () => {
    activeMainSubScopes.value = ['main-0', 'main-1'];
    const tab1 = { id: 't1', label: 'a', sessionName: 's1', ownerScope: 'main-1' as const };
    const tab2 = { id: 't2', label: 'b', sessionName: 's2', ownerScope: 'main-1' as const };
    getTerminalScope('main-1').tabs.value = [tab1 as any, tab2 as any];
    closeSubScope('main', 1);
    const main0Tabs = getTerminalScope('main-0').tabs.value;
    expect(main0Tabs.find(t => t.id === 't1')).toBeDefined();
    expect(main0Tabs.find(t => t.id === 't2')).toBeDefined();
    expect(main0Tabs.find(t => t.id === 't1')!.ownerScope).toBe('main-0');
    expect(getTerminalScope('main-1').tabs.value).toEqual([]);
  });

  it('re-points gsdTab.owningScope to scope-0 when closed scope owned it', () => {
    activeRightSubScopes.value = ['right-0', 'right-1'];
    gsdTab.value = { id: 'gsd', type: 'gsd', owningScope: 'right-1' } as any;
    closeSubScope('right', 1);
    expect(gsdTab.value!.owningScope).toBe('right-0');
  });

  it('is a no-op for index 0 (cannot close scope-0)', () => {
    activeMainSubScopes.value = ['main-0', 'main-1'];
    closeSubScope('main', 0);
    expect(activeMainSubScopes.value).toEqual(['main-0', 'main-1']);
  });

  it('persists per-project active list', () => {
    activeProjectName.value = 'projA';
    activeMainSubScopes.value = ['main-0', 'main-1'];
    layoutPatches = []; // reset spy from outer beforeEach
    closeSubScope('main', 1);
    const lastPatch = layoutPatches.at(-1);
    expect(lastPatch?.['main-active-subscopes:projA']).toBe('["main-0"]');
  });
});

**(1) `src/components/sub-scope-pane.tsx`** — append to the file (after the existing `__resetActiveSubScopesForTesting`):

```ts
import { editorTabs as _editorTabs, fileTreeTabs as _fileTreeTabs, gsdTab as _gsdTab, gitChangesTab as _gitChangesTab } from './unified-tab-bar';
// (move the existing line `import { gitChangesTab, editorTabs, fileTreeTabs, gsdTab } from './unified-tab-bar';` so we just have the singleton names available)

export function closeSubScope(zone: Zone, index: number): void {
  if (index === 0) return;
  const sig = zone === 'main' ? activeMainSubScopes : activeRightSubScopes;
  const current = sig.value;
  if (index >= current.length) return;

  const closedScope = `${zone}-${index}` as TerminalScope;
  const scope0 = `${zone}-0` as TerminalScope;

  // 1. Migrate terminal/agent tabs (keeps PTY session names stable per D-12)
  const closedTabs = getTerminalScope(closedScope).tabs.value;
  const migratedTabs = closedTabs.map(t => ({ ...t, ownerScope: scope0 }));
  getTerminalScope(scope0).tabs.value = [
    ...getTerminalScope(scope0).tabs.value,
    ...migratedTabs,
  ];
  getTerminalScope(closedScope).tabs.value = [];

  // 2. Re-point singleton ownership
  if (_gsdTab.value?.owningScope === closedScope) {
    _gsdTab.value = { ..._gsdTab.value, owningScope: scope0 };
  }
  if (_gitChangesTab.value?.owningScope === closedScope) {
    _gitChangesTab.value = { ..._gitChangesTab.value, owningScope: scope0 };
  }

  // 3. Re-point dynamic tab kinds (editor + file-tree)
  _editorTabs.value = _editorTabs.value.map(t =>
    t.ownerScope === closedScope ? { ...t, ownerScope: scope0 } : t,
  );
  _fileTreeTabs.value = _fileTreeTabs.value.map(t =>
    t.ownerScope === closedScope ? { ...t, ownerScope: scope0 } : t,
  );

  // 4. Drop the LAST scope id from the active list (fill-gap-at-end convention).
  // Scope ids are monotonic (main-0 → main-1 → main-2); since all the closed
  // scope's tabs migrated to scope-0, the list shrinks from the end regardless
  // of which index was closed.
  sig.value = current.slice(0, -1);

  // 5. Persist updates
  const project = activeProjectName.value;
  const key = activeSubScopesKey(zone, project);
  void updateLayout({ [key]: JSON.stringify(sig.value) });

  // 6. Persist tab lists for scope-0 and (now empty) closedScope
  if (project) {
    getTerminalScope(scope0).saveProjectTabs(project);
    getTerminalScope(closedScope).saveProjectTabs(project);
  }
}
```
