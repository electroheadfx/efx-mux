//! Native git status via git2 (no shell-out)

use git2::Repository;
use std::path::Path;
use tauri::async_runtime::spawn_blocking;

#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
pub struct GitStatus {
    pub branch: String,
    pub modified: usize,
    pub staged: usize,
    pub untracked: usize,
}

impl GitStatus {
    pub fn for_path(path: &str) -> Result<Self, String> {
        let repo = Repository::open(path).map_err(|e| e.to_string())?;
        let head = repo.head().map_err(|e| e.to_string())?;
        let branch = head.shorthand().unwrap_or("HEAD").to_string();

        let mut modified = 0;
        let mut staged = 0;
        let mut untracked = 0;

        let mut opts = git2::StatusOptions::new();
        opts.include_untracked(true);
        let statuses = repo.statuses(Some(&mut opts)).map_err(|e| e.to_string())?;

        for entry in statuses.iter() {
            let flags = entry.status();
            if flags.intersects(git2::Status::WT_MODIFIED) {
                modified += 1;
            }
            if flags.intersects(git2::Status::INDEX_MODIFIED | git2::Status::INDEX_NEW | git2::Status::INDEX_RENAMED) {
                staged += 1;
            }
            if flags.intersects(git2::Status::WT_NEW) {
                untracked += 1;
            }
        }

        Ok(GitStatus { branch, modified, staged, untracked })
    }
}

#[tauri::command]
pub async fn get_git_status(path: String) -> Result<GitStatus, String> {
    spawn_blocking(move || {
        GitStatus::for_path(&path)
    }).await.map_err(|e| e.to_string())?
}
```

//! Project registry commands

use crate::state::{AppState, ManagedAppState};
use tauri::async_runtime::spawn_blocking;
use tauri::State;

#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
pub struct ProjectEntry {
    pub path: String,
    pub name: String,
    pub agent: String,
    #[serde(default)]
    pub gsd_file: Option<String>,
    #[serde(default)]
    pub server_cmd: Option<String>,
}

#[tauri::command]
async fn add_project(
    state: State<'_, ManagedAppState>,
    entry: ProjectEntry,
) -> Result<(), String> {
    spawn_blocking(move || {
        let mut state = state.0.lock().map_err(|e| e.to_string())?;
        state.project.projects.push(entry);
        Ok(())
    }).await.map_err(|e| e.to_string())?
}

#[tauri::command]
async fn remove_project(
    state: State<'_, ManagedAppState>,
    name: String,
) -> Result<(), String> {
    spawn_blocking(move || {
        let mut state = state.0.lock().map_err(|e| e.to_string())?;
        state.project.projects.retain(|p| p.name != name);
        if state.project.active.as_ref() == Some(&name) {
            state.project.active = None;
        }
        Ok(())
    }).await.map_err(|e| e.to_string())?
}

#[tauri::command]
async fn switch_project(
    state: State<'_, ManagedAppState>,
    name: String,
) -> Result<(), String> {
    spawn_blocking(move || {
        let mut state = state.0.lock().map_err(|e| e.to_string())?;
        if !state.project.projects.iter().any(|p| p.name == name) {
            return Err(format!("Project '{}' not found", name));
        }
        state.project.active = Some(name);
        Ok(())
    }).await.map_err(|e| e.to_string())?
}

#[tauri::command]
async fn get_projects(
    state: State<'_, ManagedAppState>,
) -> Result<Vec<ProjectEntry>, String> {
    spawn_blocking(move || {
        let state = state.0.lock().map_err(|e| e.to_string())?;
        Ok(state.project.projects.clone())
    }).await.map_err(|e| e.to_string())?
}

#[tauri::command]
async fn get_active_project(
    state: State<'_, ManagedAppState>,
) -> Result<Option<String>, String> {
    spawn_blocking(move || {
        let state = state.0.lock().map_err(|e| e.to_string())?;
        Ok(state.project.active.clone())
    }).await.map_err(|e| e.to_string())?
}
```
