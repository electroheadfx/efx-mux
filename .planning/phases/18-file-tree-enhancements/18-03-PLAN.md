    ### 1. Add the InlineCreateRow Preact component

    ```tsx
    interface InlineCreateRowProps {
      parentDir: string;
      kind: 'file' | 'folder';
      depth: number;
      onDone: () => void;
    }

    function InlineCreateRow({ parentDir, kind, depth, onDone }: InlineCreateRowProps) {
      const [name, setName] = useState('');
      const [error, setError] = useState<string | null>(null);
      const inputRef = useRef<HTMLInputElement>(null);
      const committedRef = useRef(false);

      useEffect(() => {
        // Autofocus on mount (RESEARCH.md §5)
        inputRef.current?.focus();
      }, []);

      function validate(raw: string): string | null {
        const v = raw.trim();
        if (!v) return 'Name required';
        if (v.includes('/') || v.includes('\0')) return 'Invalid characters (no / or null)';
        return null;
      }

      async function commit() {
        if (committedRef.current) return;
        const err = validate(name);
        if (err) { setError(err); return; }
        committedRef.current = true;
        const target = `${parentDir}/${name.trim()}`;
        try {
          if (kind === 'file') await createFile(target);
          else await createFolder(target);
          onDone();
        } catch (e) {
          committedRef.current = false;
          const msg = e instanceof Error ? e.message : String(e);
          if (msg.toLowerCase().includes('already exists') || msg.toLowerCase().includes('exists')) {
            setError(`'${name.trim()}' already exists`);
          } else {
            setError(msg);
          }
          // Re-focus input so the user can fix and retry
          requestAnimationFrame(() => inputRef.current?.focus());
        }
      }

      const paddingLeft = 12 + (depth * 16);
      const Icon = kind === 'folder' ? FolderIcon : FileTextIcon;
      const placeholder = kind === 'file' ? 'New file name' : 'New folder name';

      return (
        <div>
          <div
            style={{
              padding: `${fileTreeLineHeight.value}px 12px`,
              paddingLeft,
              gap: 6,
              display: 'flex',
              alignItems: 'center',
              backgroundColor: colors.bgElevated,
            }}
          >
            <span style={{ width: 10, flexShrink: 0 }} />
            <Icon />
            <input
              ref={inputRef}
              value={name}
              placeholder={placeholder}
              aria-label={placeholder}
              aria-invalid={!!error}
              onInput={(e) => { setName((e.target as HTMLInputElement).value); setError(null); }}
              onKeyDown={(e) => {
                if (e.key === 'Enter') { e.preventDefault(); void commit(); }
                else if (e.key === 'Escape') { e.preventDefault(); committedRef.current = true; onDone(); }
              }}
              onBlur={() => { if (!committedRef.current) void commit(); }}
              style={{
                flex: 1,
                background: 'transparent',
                border: `1px solid ${error ? colors.diffRed : colors.bgBorder}`,
                borderRadius: 3,
                padding: '2px 4px',
                color: colors.textPrimary,
                fontFamily: 'var(--file-tree-font, Geist)',
                fontSize: fileTreeFontSize.value,
                outline: 'none',
              }}
              onFocus={(e) => { (e.currentTarget as HTMLInputElement).style.borderColor = colors.accent; }}
            />
          </div>
          {error && (
            <div
              style={{
                fontSize: 11,
                color: colors.diffRed,
                paddingLeft: 28 + depth * 16,
                paddingBottom: 2,
              }}
            >{error}</div>
          )}
        </div>
      );
    }
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| User right-click + keyboard → FileTree | User-initiated UI events with no untrusted input. |
| Inline input value → createFile / createFolder IPC | User types a file name. Validated client-side (non-empty, no `/` or `\0`) then forwarded to Rust, which re-validates via `is_safe_path()`. |
| count_children IPC response → confirm modal message | Rust-returned numbers interpolated into display text. No HTML — Preact renders as text. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-18-03-01 | Tampering | InlineCreateRow name → createFile/createFolder | mitigate | Client-side validation blocks `/`, `\0`, empty names. Rust-side `is_safe_path()` (from Plan 18-01) rejects `..` components. Defense in depth across JS and Rust layers. |
| T-18-03-02 | Injection | ConfirmModal title/message interpolation | mitigate | Preact interpolates `{name}` and `{count}` as text nodes, not HTML. No script injection possible even if `name` contains HTML-like syntax. |
| T-18-03-03 | Denial of Service | Delete key on tree with 0 rows | mitigate | `selectedIndex` clamp (`Math.min(selectedIndex.value, flatLen)` where `flatLen = max(0, length-1)`) ensures UI never crashes on empty. |
| T-18-03-04 | Elevation of Privilege | Keybind-triggered delete without confirmation | mitigate | Delete key opens `ConfirmModal` — same user-gated flow as the menu. No bypass. Plain `Backspace` preserves navigate-to-parent (existing safe default). |
| T-18-03-05 | Spoofing | Right-click on a row → delete different file | mitigate | `handleRowContextMenu` captures the specific `entry` at event time and passes it to `triggerDeleteConfirm(entry)`. The captured reference is what the confirm callback operates on — no race with `selectedIndex` changes. |
| T-18-03-06 | Information Disclosure | count_children result on protected folders | mitigate | count_children runs with the app's user privileges; it does not escalate. If the user selects a folder they lack read permission for, `read_dir` returns an error and the confirm message falls back to the no-count variant. |
| T-18-03-07 | Tampering | git-status-changed event storm | accept | Rust commands emit exactly once per mutation. Multiple listeners (FileTree + Git sidebar) each run `initTree/loadDir` which is idempotent. Acceptable given the single-user context. |
</threat_model>
