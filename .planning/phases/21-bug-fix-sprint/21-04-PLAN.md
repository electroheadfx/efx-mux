Refactor pattern:

```typescript
import { useRef, useEffect } from 'preact/hooks';

export function EditorTab({ tabId, filePath, fileName, content, isActive }: EditorTabProps) {
  const containerRef = useRef<HTMLDivElement>(null);
  const viewRef = useRef<EditorView | null>(null);
  const setupRef = useRef<ReturnType<typeof createEditorState> | null>(null);

  // IN-02: stash latest values in refs so the EditorView-creation useEffect
  // can read fresh callbacks/content without re-running on every render.
  // The useEffect should ONLY re-run when filePath changes (which is correct —
  // a file-path change means a different file = different EditorView instance).
  const fileNameRef = useRef(fileName);
  fileNameRef.current = fileName;
  const initialContentRef = useRef(content);
  // initialContentRef is intentionally NOT updated on each render — it captures
  // the content at first-mount/file-switch, since the useEffect uses it only at
  // EditorView creation. Subsequent content changes flow through the file-tree-changed
  // listener (Plan 21-01) which dispatches into the existing view.

  async function handleSave(docContent: string): Promise<void> {
    try {
      await writeFile(filePath, docContent);
      if (setupRef.current) setupRef.current.setSavedContent(docContent);
      setEditorDirty(tabId, false);
      showToast({ type: 'success', message: `Saved ${fileNameRef.current}` });
    } catch (err) {
      showToast({ type: 'error', message: `Save failed: ${String(err)}` });
    }
  }

  function handleDirtyChange(dirty: boolean): void {
    setEditorDirty(tabId, dirty);
  }

  // Stable refs for the callbacks so the EditorView useEffect deps array is honest.
  const handleSaveRef = useRef(handleSave);
  handleSaveRef.current = handleSave;
  const handleDirtyChangeRef = useRef(handleDirtyChange);
  handleDirtyChangeRef.current = handleDirtyChange;

  useEffect(() => {
    const container = containerRef.current;
    if (!container) return;

    const lang = getLanguageExtension(fileNameRef.current);
    const setup = createEditorState(initialContentRef.current, {
      language: lang,
      onSave: (doc) => handleSaveRef.current(doc),
      onDirtyChange: (dirty) => handleDirtyChangeRef.current(dirty),
    });

    const view = new EditorView({ state: setup.state, parent: container });
    viewRef.current = view;
    setupRef.current = setup;
    registerEditorView(tabId, view);
    registerSaveCallback(tabId, (doc) => handleSaveRef.current(doc));

    return () => {
      unregisterEditorView(tabId);
      unregisterSaveCallback(tabId);
      view.destroy();
      viewRef.current = null;
      setupRef.current = null;
    };
  }, [filePath, tabId]);
  // ↑ No eslint-disable: deps are accurate. fileName, content, and the callbacks
  //   are accessed via refs whose .current is updated on each render.
```
