**Checkbox line mapping (D-01 key insight):**
marked.js checkbox token has `{ checked: boolean }` only -- no line number. Pre-process markdown to build `checkboxIndex -> lineNumber` map before rendering.

**Algorithm:**
1. `buildLineMap(text)` -- scan markdown with `/^\s*[-*]\s*\[[ x]\]/gm` regex. Each match = checkbox at line `matchIndex + 1` (1-indexed). Returns array of line numbers.
2. `renderGSD(markdown)` -- marked.parse() with checkbox renderer returning `<input class="task-checkbox" ...>`, then post-process `<li>` tags to inject `data-line` using checkbox order.
3. `injectLineNumbers(html, lineMap)` -- walk HTML, find `<li>` tags containing `.task-checkbox`, assign `data-line` from lineMap in order.

**marked.js setup:**
```javascript

  renderer: {

    }
  }
});

  const container = html`&lt;div class="gsd-viewer"&gt;Loading...&lt;/div&gt;`;
  let currentPath = null;
  
  async function loadGSD(project) {
    if (!project || !project.gsd_file) return;
    const path = project.path + '/' + (project.gsd_file || 'PLAN.md');
    currentPath = path;
    
    const { invoke } = window.__TAURI__.core;
    const content = await invoke('read_file_content', { path });
    const lineMap = buildLineMap(content);
    const rendered = marked.parse(content);
    const withLines = injectLineNumbers(rendered, lineMap);
    container.value = `&lt;div class="gsd-content"&gt;${withLines}&lt;/div&gt;`;
  }
  
  // Listen for md-file-changed (D-03: auto-refresh)
  const { listen } = window.__TAURI__.event;
  listen('md-file-changed', () => {
    if (currentPath) {
      const project = activeProject(); // get current project
      if (project) loadGSD(project);
    }
  });
  
  // Listen for project-changed (re-render when project switches)
  document.addEventListener('project-changed', (e) =&gt; {
    const { getActiveProject } = window.__TAURI__.core;
    getActiveProject().then(name =&gt; {
      const projects = /* get from state somehow */;
      const project = projects.find(p =&gt; p.name === name);
      if (project) loadGSD(project);
    });
  });
  
  // Checkbox change handler (D-01: write-back)
  function handleClick(e) {
    if (!e.target.classList.contains('task-checkbox')) return;
    const li = e.target.closest('li');
    const line = parseInt(li.dataset.line);
    const checked = e.target.checked;
    const { invoke } = window.__TAURI__.core;
    invoke('write_checkbox', { path: currentPath, line, checked }).catch(console.error);
  }
  
  return html`
    &lt;div class="gsd-viewer" @click=${handleClick} style="height: 100%; overflow-y: auto; padding: 16px;"&gt;
      ${container}
    &lt;/div&gt;
  `;
};

import { html } from '@arrow-js/core';

export const DiffViewer = () => {
  const container = html`&lt;div class="diff-viewer" style="
    height: 100%;
    overflow-y: auto;
    padding: 8px 16px;
    font-family: 'FiraCode Light', monospace;
    font-size: 13px;
    line-height: 1.5;
  "&gt;
    &lt;div style="color: var(--text-muted);"&gt;Click a file in the sidebar to view its diff&lt;/div&gt;
  &lt;/div&gt;`;
  
  async function loadDiff(filePath) {
    const { invoke } = window.__TAURI__.core;
    try {
      const diff = await invoke('get_file_diff', { path: filePath });
      const lines = diff.split('\n');
      const highlighted = lines.map(line =&gt; {
        if (line.startsWith('+')) {
          return `&lt;div class="diff-add" style="background: rgba(133, 153, 0, 0.15); color: #859900; padding: 0 4px; margin: 0 -4px;"&gt;${escapeHtml(line)}&lt;/div&gt;`;
        } else if (line.startsWith('-')) {
          return `&lt;div class="diff-del" style="background: rgba(220, 50, 47, 0.15); color: #dc322f; padding: 0 4px; margin: 0 -4px;"&gt;${escapeHtml(line)}&lt;/div&gt;`;
        } else if (line.startsWith('@@')) {
          return `&lt;div class="diff-hunk" style="color: var(--accent); font-weight: bold; margin-top: 8px;"&gt;${escapeHtml(line)}&lt;/div&gt;`;
        } else {
          return `&lt;div class="diff-context" style="color: var(--text-muted);"&gt;${escapeHtml(line)}&lt;/div&gt;`;
        }
      }).join('');
      container.value = `&lt;pre class="diff-pre" style="margin: 0; white-space: pre-wrap; word-break: break-all;"&gt;${highlighted}&lt;/pre&gt;`;
    } catch (err) {
      container.value = `&lt;div style="color: #dc322f; padding: 16px;"&gt;Error loading diff: ${escapeHtml(err)}&lt;/div&gt;`;
    }
  }
  
  // Listen for open-diff event from sidebar.js (D-05)
  document.addEventListener('open-diff', (e) =&gt; {
    loadDiff(e.detail.path);
  });
  
  return html`&lt;div class="diff-viewer"&gt;${container}&lt;/div&gt;`;
};

function escapeHtml(text) {
  return String(text)
    .replace(/&/g, '&amp;')
    .replace(/&lt;/g, '&lt;')
    .replace(/&gt;/g, '&gt;')
    .replace(/"/g, '&quot;');
}
```

**FileTree component:**
```javascript
import { html } from '@arrow-js/core';
import { reactive } from '@arrow-js/core';

export const FileTree = (activeProject) => {
  const state = reactive({
    entries: /** @type {Array&lt;{name: string, path: string, is_dir: boolean}&gt;} */ ([]),
    selectedIndex: 0,
    currentPath: '',
    loaded: false,
  });
  
  async function loadDir(path) {
    const { invoke } = window.__TAURI__.core;
    try {
      const entries = await invoke('list_directory', { path });
      // Sort: dirs first, then files, alphabetically
      entries.sort((a, b) =&gt; {
        if (a.is_dir !== b.is_dir) return a.is_dir ? -1 : 1;
        return a.name.localeCompare(b.name);
      });
      state.entries = entries;
      state.currentPath = path;
      state.selectedIndex = 0;
      state.loaded = true;
    } catch (err) {
      console.error('[efxmux] list_directory failed:', err);
      state.entries = [];
    }
  }
  
  async function openFile(entry) {
    if (entry.is_dir) {
      loadDir(entry.path);
    } else {
      // Dispatch file-opened event for main.js to handle (D-08, PANEL-06)
      document.dispatchEvent(new CustomEvent('file-opened', { 
        detail: { path: entry.path, name: entry.name } 
      }));
    }
  }
  
  function handleKeydown(e) {
    if (!state.loaded) return;
    switch (e.key) {
      case 'ArrowDown':
        e.preventDefault();
        state.selectedIndex = Math.min(state.selectedIndex + 1, state.entries.length - 1);
        break;
      case 'ArrowUp':
        e.preventDefault();
        state.selectedIndex = Math.max(state.selectedIndex - 1, 0);
        break;
      case 'Enter':
        e.preventDefault();
        if (state.entries[state.selectedIndex]) {
          openFile(state.entries[state.selectedIndex]);
        }
        break;
      case 'Backspace':
        // Go up one directory level
        e.preventDefault();
        const parent = state.currentPath.split('/').slice(0, -1).join('/');
        if (parent) loadDir(parent);
        break;
    }
  }
  
  // Initial load with active project path
  if (activeProject &amp;&amp; activeProject.path) {
    loadDir(activeProject.path);
  }
  
  // Re-load on project change
  document.addEventListener('project-changed', (e) =&gt; {
    // Find project path from state
    const { invoke } = window.__TAURI__.core;
    invoke('get_active_project').then(name =&gt; {
      invoke('get_projects').then(projects =&gt; {
        const project = projects.find(p =&gt; p.name === name);
        if (project) loadDir(project.path);
      });
    });
  });
  
  return html`
    &lt;div 
      class="file-tree" 
      tabindex="0" 
      @keydown=${handleKeydown}
      style="
        height: 100%;
        overflow-y: auto;
        padding: 4px 0;
        outline: none;
        background: var(--bg-base);
      "
    &gt;
      ${() =&gt; state.entries.length === 0 ? html`
        &lt;div style="padding: 16px; color: var(--text-muted); font-size: 13px;"&gt;
          ${state.loaded ? 'Empty directory' : 'Loading...'}
        &lt;/div&gt;
      ` : state.entries.map((entry, i) =&gt; html`
        &lt;div
          class="${() =&gt; i === state.selectedIndex ? 'selected' : ''}"
          style="
            display: flex;
            align-items: center;
            padding: 4px 12px;
            cursor: pointer;
            font-size: 13px;
            color: ${() =&gt; i === state.selectedIndex ? 'var(--text-bright)' : 'var(--text)'};
            background: ${() =&gt; i === state.selectedIndex ? 'var(--bg-raised)' : 'transparent'};
            gap: 6px;
          "
          @click=${() =&gt; { state.selectedIndex = i; openFile(entry); }}
          @mouseenter=${() =&gt; { state.selectedIndex = i; }}
        &gt;
          &lt;span style="color: var(--accent); font-size: 11px; width: 14px; flex-shrink: 0;"&gt;
            ${entry.is_dir ? '/' : ' '}
          &lt;/span&gt;
          &lt;span style="overflow: hidden; text-overflow: ellipsis; white-space: nowrap;"&gt;
            ${entry.name}
          &lt;/span&gt;
        &lt;/div&gt;
      `)}
    &lt;/div&gt;
  `;
};
```
