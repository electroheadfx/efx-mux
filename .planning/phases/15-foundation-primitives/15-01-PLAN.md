```typescript
export interface DropdownItem {
  label: string;
  action: () => void;
  icon?: preact.ComponentType<{ size?: number }>;
  disabled?: boolean;
  separator?: boolean;
}

export interface DropdownProps {
  items: DropdownItem[];
  trigger: (props: {
    onClick: () => void;
    'aria-haspopup': 'menu';
    'aria-expanded': boolean;
  }) => preact.VNode;
}
```

2. Export Dropdown component (D-04: uncontrolled state):
   - Use signal for isOpen (internal, not exported)
   - Use signal for selectedIndex (tracks focused item)
   - Render prop trigger pattern (D-06): call props.trigger with onClick/aria props
   
3. Keyboard navigation (D-05):
   - ArrowDown: increment selectedIndex (clamp to max)
   - ArrowUp: decrement selectedIndex (clamp to 0)
   - Home: set selectedIndex to 0
   - End: set selectedIndex to last selectable item
   - Enter/Space: call items[selectedIndex].action(), close menu
   - Escape: close menu
   - Type-ahead: buffer characters for 500ms, find first item starting with buffer

4. Focus management:
   - Set tabIndex={-1} on menu container
   - Focus menu container when opened (use useEffect + ref.focus())
   - Restore focus to trigger on close (track trigger ref)

5. Styling:
   - position: absolute, top: 100%, left: 0
   - Same bgElevated, bgBorder, radii.lg as ContextMenu
