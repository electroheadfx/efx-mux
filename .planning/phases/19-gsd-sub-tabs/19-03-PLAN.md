    ```typescript
    import { colors, fonts, fontSizes } from '../../tokens';

    export interface StatusBadgeProps {
      status: 'complete' | 'in-progress' | 'not-started';
    }

    export function StatusBadge({ status }: StatusBadgeProps) {
      const glyph = status === 'complete' ? '✓' : status === 'in-progress' ? '◆' : '○';
      const color = status === 'complete' ? colors.statusGreen
        : status === 'in-progress' ? colors.statusYellow
        : colors.textMuted;
      const label = status === 'complete' ? 'Complete'
        : status === 'in-progress' ? 'In progress' : 'Not started';
      return (
        <span aria-label={label} title={label} style={{
          fontFamily: fonts.sans, fontSize: fontSizes.base, color, flexShrink: 0,
        }}>
          {glyph}
        </span>
      );
    }
    ```
