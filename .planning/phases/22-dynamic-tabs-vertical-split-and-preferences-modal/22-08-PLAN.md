```ts
describe('Phase 22 gap-closure (22-08): per-mutation persistence', () => {
  let savedSessions: Array<Record<string, string>> = [];

  beforeEach(async () => {
    savedSessions = [];
    // reset signals + scope state per existing pattern
    activeProjectName.value = 'projZ';
    mockIPC((cmd, args) => {
      if (cmd === 'load_state') {
        return {
          version: 1,
          layout: {},
          theme: { mode: 'dark' },
          session: {},
          project: { active: 'projZ', projects: [{ name: 'projZ', path: '/tmp/projZ' }] },
          panels: {},
        };
      }
      if (cmd === 'save_state') {
        const state = JSON.parse((args as any).stateJson);
        savedSessions.push(state.session ?? {});
        return undefined;
      }
      if (cmd === 'spawn_terminal') return undefined;
      if (cmd === 'kill_pty') return undefined;
    });
    await loadAppState();
    // Reset main-0 + right-0 tabs
    getTerminalScope('main-0').tabs.value = [];
    getTerminalScope('right-0').tabs.value = [];
  });

  it('createNewTab on main-0 persists terminal-tabs:projZ:main-0', async () => {
    await getTerminalScope('main-0').createNewTab();
    const last = savedSessions.at(-1)!;
    expect(last['terminal-tabs:projZ:main-0']).toBeDefined();
    const arr = JSON.parse(last['terminal-tabs:projZ:main-0']);
    expect(Array.isArray(arr) ? arr.length : (arr.tabs?.length ?? 0)).toBeGreaterThanOrEqual(1);
  });

  it('closeTab on main-0 persists reduced terminal-tabs:projZ:main-0', async () => {
    await getTerminalScope('main-0').createNewTab();
    await getTerminalScope('main-0').createNewTab();
    expect(getTerminalScope('main-0').tabs.value).toHaveLength(2);
    const idToClose = getTerminalScope('main-0').tabs.value[0].id;
    await getTerminalScope('main-0').closeTab(idToClose);
    const last = savedSessions.at(-1)!;
    const persisted = JSON.parse(last['terminal-tabs:projZ:main-0']);
    const persistedTabs = Array.isArray(persisted) ? persisted : (persisted.tabs ?? []);
    expect(persistedTabs.length).toBe(1);
    expect(persistedTabs.find((t: any) => t.id === idToClose)).toBeUndefined();
  });

  it('allocateNextSessionName persists tab-counter:projZ', async () => {
    await getTerminalScope('main-0').createNewTab();
    await getTerminalScope('main-0').createNewTab();
    await getTerminalScope('main-0').createNewTab();
    const last = savedSessions.at(-1)!;
    expect(last['tab-counter:projZ']).toBe('3');
  });
});
```
