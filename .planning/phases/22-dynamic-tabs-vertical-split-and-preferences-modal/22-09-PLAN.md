    expect(tabEl!.textContent).not.toContain('Git Changes');

    fileTreeTabs.value = [{ id: 'ft-r0', type: 'file-tree', ownerScope: 'right-0' } as any];
    render(<UnifiedTabBar scope="right-0" />);
    const tabEl = document.querySelector('[data-tab-id="ft-r0"]');
    expect(tabEl).not.toBeNull();
    expect(tabEl!.textContent).toContain('File Tree');
    expect(tabEl!.textContent).not.toContain('Git Changes');
  });

  it('GSD tab close button clears gsdTab signal', async () => {
    gsdTab.value = { id: 'gsd', type: 'gsd', owningScope: 'right-0' } as any;
    render(<UnifiedTabBar scope="right-0" />);
    const tabEl = document.querySelector('[data-tab-id="gsd"]')!;
    const closeBtn = tabEl.querySelector('[aria-label*="Close" i], .tab-close-btn, button') as HTMLElement;
    expect(closeBtn).not.toBeNull();
    closeBtn.click();
    expect(gsdTab.value).toBeNull();
  });

  it('File Tree tab close button removes from fileTreeTabs', async () => {
    fileTreeTabs.value = [{ id: 'ft-r0', type: 'file-tree', ownerScope: 'right-0' } as any];
    render(<UnifiedTabBar scope="right-0" />);
    const tabEl = document.querySelector('[data-tab-id="ft-r0"]')!;
    const closeBtn = tabEl.querySelector('[aria-label*="Close" i], .tab-close-btn, button') as HTMLElement;
    closeBtn.click();
    expect(fileTreeTabs.value).toHaveLength(0);
  });

  it('GSD tab element has user-select: none', () => {
    gsdTab.value = { id: 'gsd', type: 'gsd', owningScope: 'right-0' } as any;
    render(<UnifiedTabBar scope="right-0" />);
    const tabEl = document.querySelector('[data-tab-id="gsd"]') as HTMLElement;
    // jsdom does not always compute custom CSS â€” check inline first, fall back to computed
    const userSelect = tabEl.style.userSelect || getComputedStyle(tabEl).userSelect;

  });

  it('handleCrossScopeDrop accepts gsd into empty target pane', () => {

    handleCrossScopeDrop('gsd', 'right-0', '', 'main-1', false);
    expect(gsdTab.value!.owningScope).toBe('main-1');
  });
});
