```tsx
import { useEffect, useRef } from 'preact/hooks';
import type { ComponentType } from 'preact';
import { colors, radii, spacing, fonts } from '../tokens';

export interface ContextMenuItem {
  label: string;
  action: () => void;
  icon?: ComponentType<{ size?: number }>;
  disabled?: boolean;
  separator?: boolean;
}

export interface ContextMenuProps {
  items: ContextMenuItem[];
  x: number;
  y: number;
  onClose: () => void;
}

// Auto-flip (lines 28-35):
useEffect(() => {
  if (!menuRef.current) return;
  const rect = menuRef.current.getBoundingClientRect();
  const flipX = x + rect.width > window.innerWidth;
  const flipY = y + rect.height > window.innerHeight;
  if (flipX) menuRef.current.style.left = `${x - rect.width}px`;
  if (flipY) menuRef.current.style.top = `${y - rect.height}px`;
}, [x, y]);

// Close on outside click + Escape (lines 38-55):
useEffect(() => {
  const handleClickOutside = (e: MouseEvent) => {
    if (menuRef.current && !menuRef.current.contains(e.target as Node)) onClose();
  };
  const handleEscape = (e: KeyboardEvent) => { if (e.key === 'Escape') onClose(); };
  document.addEventListener('mousedown', handleClickOutside);
  document.addEventListener('keydown', handleEscape);
  return () => {
    document.removeEventListener('mousedown', handleClickOutside);
    document.removeEventListener('keydown', handleEscape);
  };
}, [onClose]);
```

    ```tsx
      key={i}
    ```

    ```tsx
      const handleArrowKeys = (e: KeyboardEvent) => {

          // Find the currently hovered row (submenuIndex target) — only open if it has children
          const focusedIdx = submenuIndex;
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| User DOM input → ContextMenu | Right-click events from user interactions. No untrusted string input — the menu items array is constructed by Efxmux code, not arbitrary DOM. |
| tauri.conf.json → WKWebView | `dragDropEnabled: true` enables the OS-level drop event pipeline. Paths arrive from the OS (Finder), not the user's JS. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-18-02-01 | Tampering | ContextMenu children rendering | mitigate | Recursive render uses the same `ContextMenu` component for submenus — all existing guards (click-outside, Escape, auto-flip) inherit. No new DOM sanitization needed because `label` is already rendered as text (not innerHTML). |
| T-18-02-02 | Information Disclosure | Hover delay timer | accept | 150ms timer state is local to component; no side channels. React-style re-render is the only observable effect. |
| T-18-02-03 | Elevation of Privilege | dragDropEnabled | mitigate | Setting `dragDropEnabled: true` only enables the EVENT. The actual handling in downstream plan 18-05 applies an `is_safe_path` guard in Rust for every dropped path. No privilege added at config time. |
| T-18-02-04 | Denial of Service | Submenu infinite recursion | mitigate | Although `ContextMenuItem.children: ContextMenuItem[]` is recursive, menus are constructed statically in calling code (file-tree.tsx) — no runtime cycle. Worst case is 2 levels (parent + submenu) in Phase 18; depth is bounded by the menu builder code, not this component. |
| T-18-02-05 | Tampering | tauri.conf.json | accept | Config file is under version control and packaged into the app bundle — not user-editable at runtime. Changes require a rebuild. |
